public string SwitchStatementGoto(int value)
    {
        string ret;
        switch (value)
        {
            case 0:
                goto case 1;

            case 1:
                ret = "ZeroOrOne";
                break;

            case 2:
                goto default;

            default:
                ret = "Other";
                break;
        }
        return ret;
    }
