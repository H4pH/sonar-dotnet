public void DoTryBreakCatchContinueFinally()
    {
        do
        {
            A();
            try
            {
                B();
                break;
                C();
            }
            catch(Exception ex)
            {
                Handle(ex);
                continue;
            }
            finally
            {
                F();
            }
            D();
        } while (true);
    }
