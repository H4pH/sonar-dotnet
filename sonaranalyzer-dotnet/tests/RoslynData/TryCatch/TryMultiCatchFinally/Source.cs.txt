public void TryMultiCatchFinally()
    {
        try
        {
            A();
            B();
            C();
        }
        catch (FormatException ex)
        {
            Handle(ex);
        }
        catch (ArgumentNullException ex)
        {
            Handle(ex);
        }
        catch (Exception ex) when (ex.Message.Contains("memory"))
        {
            Handle(ex);
        }
        catch (Exception ex)
        {
            Handle(ex);
        }
        finally
        {
            F();
        }
    }
