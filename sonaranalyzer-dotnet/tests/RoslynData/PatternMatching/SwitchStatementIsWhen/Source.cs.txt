public string SwitchStatementIsWhen(object arg)
    {
        switch (arg)
        {
            case TypeA a when a.ValueA == 0:
                return a.ToString();
            case TypeB b when b.ValueB == 0:
                return b.ToString();
            case TypeC c when c.ValueC == 0:
                return c.ToString();
            default:
                return null;
        }
    }
